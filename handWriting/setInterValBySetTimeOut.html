    setTimeout 实现 setInterval

    # 为什么要用setTimeout实现setInterval

    setInterval 缺点

    • 使用 setInterval 时，某些间隔会被跳过；即使 setInterval 调用的方法报错了，他仍然会继续执行。
    • 无视网络延迟，可能多个定时器会连续执行；

    # 总结

    每个 setTimeout 产生的任务会直接 push 到任务队列中；而 setInterval 在每次把任务 push 到任务队列前，都要进行一下判断(看上次的任务是否仍在队列中)。

    因而我们一般用 setTimeout 模拟 setInterval，来规避掉上面的缺点。

    /**
 * 
 *
 * @param {*} fn 执行的函数
 * @param {*} delay 延时
 * @param {boolean} [immediate=false] 是否立即执行
 * @return {*} 
 */
const mySetInterVal = (fn, delay, immediate = false) => {
    let time = null;
    function run() {
       time =  setTimeout(() => {
           fn()
           run();
        }, delay)
    }

    if(immediate) fn()
    run();

    return {
        clear: () => {
            clearTimeout(time)
        }
    }
}
const d = +new Date();

const t = mySetInterVal(() => console.log(+new Date() - d), 1000, true);

setTimeout(() => {
    t.clear()
}, 5000);
